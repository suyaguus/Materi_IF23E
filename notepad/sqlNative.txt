CREATE DATABASE db_penilaian;
USE db_penilaian;
CREATE TABLE table_mahasiswa(
NPM CHAR (8) PRIMARY KEY,
Nama VARCHAR (25),
Alamat VARCHAR (250),
Usia INT
);

CREATE TABLE table_matakuliah(
Kode_Matkul CHAR (6) PRIMARY KEY,
Nama_Matkul VARCHAR (25),
SKS CHAR (1),
Dosen VARCHAR (25)
);

CREATE TABLE table_detail_penilaian(
NPM CHAR (8),
Kode_Matkul CHAR (6),
Nilai CHAR (1),
FOREIGN KEY (NPM) REFERENCES table_mahasiswa(NPM),
FOREIGN KEY (Kode_Matkul) REFERENCES table_matakuliah(Kode_Matkul)
);

foreign key tidak menggunakan primary key tapi terhubung dengan table lainnya.
